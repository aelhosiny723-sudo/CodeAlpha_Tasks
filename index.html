const flashcards = [

  { question: "ما هو العضو المسؤول عن ضخ الدم في جسم الإنسان؟", answer: "القلب  " },

  { question: "ما هي وحدة قياس المقاومة الكهربائية؟", answer: "الأوم (Ohm)  " },

  { question: "كم عدد كواكب المجموعة الشمسية؟", answer: "8 كواكب  " }

];

 

let currentIndex = 0;

 

function flipCard() {

  document.querySelector('.card').classList.toggle('flipped');

}

 

function nextCard() {

  const card = document.querySelector('.card');

  if (card.classList.contains('flipped')) {

    card.classList.remove('flipped');

    setTimeout(changeContent, 300);

  } else {

    changeContent();

  }

}

 

function changeContent() {

  currentIndex = (currentIndex + 1) % flashcards.length;

  document.querySelector('.card-front p').innerText = flashcards[currentIndex].question;

  document.querySelector('.card-back p').innerText = flashcards[currentIndex].answer;

}

function newQuote() {
